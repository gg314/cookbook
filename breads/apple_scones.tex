\section{Apple Scones with Icing}
\begin{recipe}
	\yieldtime{? servings}{? minutes}

	\pre{...}

	\ingredientlist{
		\header{For the scones:}
		
		2\sfrac{1}{2} \cup flour

		\sfrac{1}{3} \cup sugar

		1 \T baking powder

		1 \t baking soda

		\sfrac{1}{2} \t salt

		6 \T butter, cold

		1\sfrac{1}{4} \cup heavy whipping cream

		1 \cup Honeycrisp apple, diced
		
		\header{For the icing:}
		
		5 \T butter, softened
			
		\sfrac{1}{8} \t apple pie spice
			
		1 \cup powdered sugar
			
		2 \T heavy cream
	}

	Preheat oven to 425\degree{F}.

	In a small bowl, whisk together flour, sugar, baking powder, baking soda, and salt. Using a pastry blender, cut in the cold butter to form a crumb-like mixture. Stir in the cream and gently fold in the apples.

	Knead the dough on a well-floured surface. Add up to \sfrac{1}{4} cup extra flour if dough is still sticky. Roll out dough to at least a 1" thickness.

	Cut into triangular pieces and lay on a greased baking sheet or parchment paper. Bake for 14 minutes or until golden brown.

	\smallheader{For the icing: } In a small bowl, combine the icing ingredients. Use a hand mixer and blend until smooth. Drizzle over scones when cool.

\end{recipe}
