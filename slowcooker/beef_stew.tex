\section{Provencal Beef Stew}
\begin{recipe}
	\yieldsourcetime{}{Slow-Cooker Fall Favorites}{}

\pre{You want stew beef with a high fat content, 80 or 85 percent.}

\ingredientlist{
	1 tablespoon oil \\
	1 lb. stew beef \\
	1\sfrac{1}{2} c. red wine or beef broth \\
	2 carrots, peeled and thickly sliced \\
	6 small potatoes (optional), quartered \\
	1 onion, coarsely chopped \\
	14\sfrac{1}{2} oz. can diced tomatoes, drained \\
	\sfrac{1}{2} T. tomato paste \\
	2 cloves garlic, minced \\
	\sfrac{1}{2} t. dried thyme \\
	\sfrac{1}{2} pepper \\
	\sfrac{1}{4} orange zest \\
	\sfrac{1}{4} mace \\
	2 t. fresh parsley, chopped \\
}

In a skillet over medium heat, brown beef cubes on all sides; drain.

Combine beef and remaining ingredients in a slow cooker. Cover and
cook on low setting for 7 to 8 hours, until beef is tender and stew is
hot.

\end{recipe}
