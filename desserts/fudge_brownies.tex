\section{Fudge Brownies with Frosting}
\begin{recipe}
	\yieldtime{9 $\times$ 13" pan}{45 minutes}

	\pre{``The best'' fudge brownies.}

	\ingredientlist{
		\sfrac{1}{2} \cup margarine or butter

		4 eggs

		2 \cup sugar

		1 \cup flour

		\sfrac{2}{3} \cup unsweetened cocoa

		2 \t vanilla

		\sfrac{1}{4} \t baking powder

		\sfrac{1}{4} \t salt

		1 \cup nuts (optional)

		\header{Frosting:}

		\sfrac{1}{2} \cup cocoa

		2 \sfrac{2}{3} \cup powdered sugar

		6 \T butter

		1 \t vanilla

		1 \T corn syrup (optional)
	}

	Melt butter in double-boiler. Add sugar. Beat in eggs one at a time. Add vanilla.

	In a bowl, combine flour, cocoa, salt, and baking powder. Pour sugar mixture into dry ingredients; mix well. Pour into greased 9$\times$13" pan. Bake at 350\degree{F} for 20-25 minutes.

	\smallheader{Frosting: } Combine \sfrac{1}{2} \cup cocoa with 2 \sfrac{2}{3} \cup powdered sugar. Cream 6 \T butter with \sfrac{1}{2} \cup of the cocoa mixture. Add remaining cocoa mixture, alternating with a little milk. Beat to spreading consistency. Add 1 \t vanilla and 1 \T corn syrup (optional for glossier frosting). Makes two cups.

\end{recipe}
