\section{Samosadillas}
\begin{recipe}
	\yieldsourcetime{4-6 portions}{FoodWishes.com}{45 minutes}

	\pre{A classic samosa filling without a deep-fryer. Instead, a quesadilla-like form factor gives the familiar flavor with much less mess.}

	\ingredientlist{
		\header{Samosa quesadillas:}

		6 large flour tortillas

		1 \T olive oil

		1 \T butter

		1 large yellow onion, diced

		3 cloves garlic, finely chopped

		1 Serrano pepper, minced

		1 \sfrac{1}{2} \tsp fresh ginger, finely chopped

		1 \tsp kosher salt plus more to taste

		1 \sfrac{1}{2} \tsp garam masala

		1 \tsp ground cumin

		\sfrac{1}{2} \tsp ground coriander

		\sfrac{1}{2} \tsp turmeric

		\sfrac{1}{4} \tsp cayenne

		1 \cup green peas, thawed and drained

		1 \sfrac{1}{2} pounds Yukon gold potatoes, peeled

		2 \T chopped cilantro

		1 lemon

		\header{Cilantro lime chutney:}

		1 large bunch fresh cilantro

		\sfrac{1}{3} \cup mint leaves

		\sfrac{1}{4} \cup green onions, sliced

		1 Serrano pepper, sliced

		1 or 2 limes, juiced

		\sfrac{1}{2} cup plain yogurt

		pinch of salt, optional
	}

	Cut potatoes in $1/4"$ cubes, transfer into cold water, and bring up to a simmer with a large spoon of salt. Back heat down to medium until potatoes are nice and soft. Drain and set aside.

	Saut/'{e} diced onion with butter and olive oil until softened and translucent. In a bowl, create spice mix with kosher salt, garam masala, cumin, coriander, turmeric, and cayenne.

	Add to the onions: minced chili, garlic, ginger root, and spice blend. Cook and stir for two minutes, add peas. Continue cooking until peas are firm but softened, about 3 minutes.

	In a mixing bowl, combine potatoes, onions and peas, a handful of chopped cilantro, and the juice of a lemon. Mix with a spatula, half-mashing the potatoes. Taste and consider adding more salt.

	\smallheader{For the sauce: } Wash cilantro. In a blender, add chopped cilantro, mint, green onion, chili, lime juice, and yogurt. Blend.

	\smallheader{Cooking the samosadilla: } Cover half of tortilla with potato filling, fold over, and brown with olive oil in a pan for 2-3 minutes per side.

\end{recipe}
