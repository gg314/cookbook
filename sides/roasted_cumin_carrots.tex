\section{Roasted Cumin Carrots}
\begin{recipe}
	\yieldtime{4-6 servings}{15 min. prep, 60 min. roast}

	\pre{Inspired by an Israeli \textit{mezze} (appetizer), you can substitute almost any root vegetable, such as potatoes, parsnips, turnips, or rutabagas, for the carrots.}

	\ingredientlist{
		3 \lbs carrots, peeled and cut into large chunks

		2 \Tbsp olive oil

		2 \T ground cumin

		2 \tsp kosher salt

		1 \tsp black pepper

		\sfrac{1}{4} \cup chopped fresh cilantro leaves (plus additional for garnish)

		3 garlic cloves, finely chopped

		\sfrac{1}{2} \cup crumbled feta cheese
	}


	Preheat the oven to 400\degree{F}.

	Place all of the ingredients except the feta in a large bowl and mix well. Spread on a large ungreased baking sheet. Roast until the carrots are soft inside and crisp and browned on the outside, about 45 minutes to 1 hour.

	Add the feta cheese, toss gently, and garnish with cilantro. Serve hot or at room temperature.



\end{recipe}
