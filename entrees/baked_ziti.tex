\section{Baked Ziti}
\begin{recipe}
	\yieldtime{6-7 servings}{75-90 minutes}

	\pre{Doubling the proportions will overflow a $9"\times13"$ baking pan, but you can just as well bake two dishes at once!}

	\ingredientlist{
		\header{Start pasta:}

		1 \lb ziti pasta

		\header{Everything else:}

		1 onion, chopped

		2 garlic cloves, minced

		16 oz. can tomato sauce or diced tomatoes

		1 \cup ricotta cheese

		8 oz. shredded mozarella cheese

		\sfrac{1}{4} \cup parmesan cheese

		\sfrac{1}{2} \cup parsley

		\sfrac{1}{2} \T dried basil

		\sfrac{1}{2} \T dried oregano

		\sfrac{1}{2} \t black pepper

		\sfrac{1}{4} \t red pepper flakes

		\sfrac{1}{4} \t salt

		\sfrac{1}{4} \cup olive oil

		\sfrac{1}{4} \cup breadcrumbs
	}

	Preheat oven to 400$^{\circ}$F.

	Cook pasta to al dente. Coat baking dish with oil.

	In a skillet, heat the remaining oil and cook the onion until translucent. Add garlic, and cook 1 min. Add tomato sauce and cook 3-5 minutes.

	In a large bowl, mix mozarella and ricotta cheeses, spices, and herbs. When combined, add sauce mixture and stir.

	Add pasta and toss well. Spoon into baking dish. Sprinkle breadcrumbs and parmesan on top. Bake at 400$^{\circ}$F for 20-30 minutes. Allow 5 minutes to rest before serving.

\end{recipe}
